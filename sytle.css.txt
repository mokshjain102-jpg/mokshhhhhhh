let history = [];

// SCAN
function analyze(){
let file = document.getElementById("file").files[0];
if(!file){alert("Select image"); return;}

let diseases=["Blight","Leaf Spot","Healthy"];
let result = diseases[Math.floor(Math.random()*3)];

let text = result + " detected";
document.getElementById("output").innerText = text;

history.push(text);
}

// CHAT
function sendMsg(e){
if(e.key==="Enter"){
let msg=document.getElementById("chatInput").value;
handleBot(msg);
}
}

function handleBot(msg){
let reply="Upload image for better result";

if(msg.includes("blight")) reply="Use fungicide";
if(msg.includes("yellow")) reply="Add fertilizer";

addMsg("You: "+msg);
addMsg("Bot: "+reply);

// voice reply
let speech=new SpeechSynthesisUtterance(reply);
speech.lang="hi-IN";
speechSynthesis.speak(speech);
}

// VOICE INPUT
function startVoice(){
let rec=new (window.SpeechRecognition||window.webkitSpeechRecognition)();
rec.lang="hi-IN";
rec.start();

rec.onresult=function(e){
let text=e.results[0][0].transcript;
handleBot(text);
}
}

// UI
function addMsg(text){
let box=document.getElementById("chatBox");
let div=document.createElement("div");
div.innerText=text;
box.appendChild(div);
}
